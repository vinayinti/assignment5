{
    int i,j,k=0;
    for(i=0;i<n;i++)

        for(j=i+1;j<n;j++)
        {

            {
               int t;
               t=*(p+i);
            *(p+i)=*(p+j);
               *(p+j)=t; 
            }
        }
    }
